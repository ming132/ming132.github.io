#include<iostream>
#include<algorithm>
#include<string.h>
#include<iomanip>
#include<map>
#define INF 0x3f3f3f3f
using namespace std;
int mem[5][5],o1,o2,x,y,ant;
map<int,char> q;
void print()
{
	cout<<"当前棋盘\n\n";
	cout<<"  "<<1<<" "<<2<<" "<<3<<"\n";
	for(int i=1;i<4;i++){
	cout<<i<<" ";
	for(int j=1;j<4;j++)
	cout<<q[mem[i][j]]<<" ";
	cout<<"\n";
}
}
bool empty_()
{
	for(int i=1;i<4;i++)
	for(int j=1;j<4;j++)
	if(mem[i][j]==0)
	return 0;
	return 1;
}
bool safe(int u,int v)
{
	if(0<u&&u<4&&0<v&&v<4)
	return 1;
	return 0;
}
int val()
{
	int cnt=0;
	for(int i=1;i<4;i++)
	{
		for(int j=1;j<4;j++)
		{
			if(mem[i][j]==-1&&mem[i+1][j]==-1)
			{
				if(safe(i+2,j)&&mem[i+2][j]==0)
				{
					cnt+=1000;
				}
				else
				cnt+=10;
			}
			if(mem[i][j]==-1&&mem[i][j+1]==-1)
			{
				if(safe(i,j+2)&&mem[i][j+2]==0)
				{
					cnt+=1000;
				}
				else
				cnt+=10;
			}
			if(mem[i][j]==1&&mem[i+1][j]==1)
		    {
				if(safe(i+2,j)&&mem[i+2][j]==0)
				{
					cnt-=1000;
				}
				else
				cnt-=10;
			}
			if(mem[i][j]==1&&mem[i][j+1]==1)
			{
				if(safe(i,j+2)&&mem[i][j+2]==0)
				{
					cnt-=1000;
				}
				else
				cnt-=10;
			}
			if(mem[i][j]==-1&&mem[i+1][j+1]==-1)
			{
				if(safe(i+2,j+2)&&mem[i+2][j+2]==0)
				{
					cnt+=1000;
				}
				else
				cnt+=10;
			}
			if(mem[i][j]==1&&mem[i+1][j+1]==1)
			{
				if(safe(i+2,j+2)&&mem[i+2][j+2]==0)
				{
					cnt-=1000;
				}
				else
				cnt-=10;
			}
			if(mem[i][j]==-1&&mem[i-1][j+1]==-1)
			{
				if(safe(i-2,j+2)&&mem[i-2][j+2]==0)
				{
					cnt+=1000;
				}
				else
				cnt+=10;
			}
			if(mem[i][j]==1&&mem[i-1][j+1]==1)
			{
				if(safe(i-2,j+2)&&mem[i-2][j+2]==0)
				{
					cnt+=1000;
				}
				else
				cnt+=10;
			}
		}
	}
	return cnt;
}
int win()
{
	if(mem[1][1]+mem[1][2]+mem[1][3]==3)
	return 1;
	if(mem[2][1]+mem[2][2]+mem[2][3]==3)
	return 1;
	if(mem[3][1]+mem[3][2]+mem[3][3]==3)
	return 1;
	if(mem[1][1]+mem[2][1]+mem[3][1]==3)
	return 1;
	if(mem[1][2]+mem[2][2]+mem[3][2]==3)
	return 1;
	if(mem[1][3]+mem[2][3]+mem[3][3]==3)
	return 1;
	if(mem[1][1]+mem[2][2]+mem[3][3]==3)
	return 1;
	if(mem[1][3]+mem[2][2]+mem[3][1]==3)
	return 1;
	if(mem[1][1]+mem[1][2]+mem[1][3]==-3)
	return -1;
	if(mem[2][1]+mem[2][2]+mem[2][3]==-3)
	return -1;
	if(mem[3][1]+mem[3][2]+mem[3][3]==-3)
	return -1;
	if(mem[1][1]+mem[2][1]+mem[3][1]==-3)
	return -1;
	if(mem[1][2]+mem[2][2]+mem[3][2]==-3)
	return -1;
	if(mem[1][3]+mem[2][3]+mem[3][3]==-3)
	return -1;
	if(mem[1][1]+mem[2][2]+mem[3][3]==-3)
	return -1;
	if(mem[1][3]+mem[2][2]+mem[3][1]==-3)
	return -1;
	return 0;
}
int al_be(int h,int player,int a,int b)
{
	if(h==15||win()!=0){
	if(win()==1)
	{
		return -100000;
	}
	if(win()==-1)
	{
		return 100000;
	}
		return val();
	}
	if(empty_()){
	return 0;
    }
//	cout<<player<<endl;
	if(player==1)
	{
		for(int i=1;i<4;i++)
		{
			for(int j=1;j<4;j++)
			{
				if(mem[i][j]==0)
				{
					mem[i][j]=-1;
					//cout<<1<<" "<<i<<j<<" "<<a<<" "<<b<<endl;
					int ans=al_be(h+1,player^1,a,b);
					//cout<<ans<<endl;
					mem[i][j]=0;
					if(ans>a)
					{
						a=ans;
						x=i,y=j;
					}
					if(a>=b)
					{
						return a;
					}
					//cout<<a<<" "<<b<<endl;
				}
			}
		}
		return a;
	}
	else
	{
		for(int i=1;i<4;i++)
		{
			for(int j=1;j<4;j++)
			{
				if(mem[i][j]==0)
				{
					mem[i][j]=1;
					//cout<<0<<" "<<i<<j<<" "<<a<<" "<<b<<endl;
					int ans=al_be(h+1,player^1,a,b);
					//cout<<"ans: "<<ans<<endl;
					mem[i][j]=0;
					if(ans<b)
					{
						b=ans;
					}
					if(a>=b)
					{
						return b;
					}
				}
			}
		}
		return b;
	}
}
void decide()
{
	cout<<"让我想想该怎么下\n";
	cout<<"想好了！\n";
	al_be(0,1,-INF,INF);
	cout<<"我下在: "<<x<<" "<<y<<"\n";
	mem[x][y]=-1;
}
int main()
{
	int t;
	memset(mem,0,sizeof(mem));
	q[0]='?';
	q[-1]='x';
	q[1]='o';
	cout<<setw(50)<<"欢迎来到简单的井字棋游戏\n\n\n";
	cout<<"请选择先手or后手，输入0表示自己先手，输入1表示电脑先手\n请输入：";
	cin>>t; 
	if(t!=1&&t!=0)
	{
		cout<<"输入有误，默认玩家先手\n\n";
		t=0; 
	}
	while((!empty_())&&(!win()))
	{
	//	ant++;
	//	if(ant>2)break;
	//	cout<<ant<<endl;
		print();
		if(t==0)
		{
			while(1)
			{
				cout<<"请输入您的落子位置，您的子为o，格式为1 1\n";
				cin>>o1>>o2;
				if(mem[o1][o2]!=0)
				{
					cout<<"输入有误，请重新输入：";
					continue;
				}
				mem[o1][o2]=1;
				t=t^1;
				break;
			}
		}
		print();
		if(win()==1)
		{
			cout<<"你赢了，不过这是不存在的！";
			return 0; 
		}
		if(empty_())
		{
			cout<<"平局！你是赢不了我的！";
			return 0;
		}
		if(t==1)
		{
			decide();
			t=t^1;
		}
		if(empty_())
		{
			cout<<"平局！你是赢不了我的！";
			return 0;
		}
		if(win()==-1)
		{
			cout<<"我赢了，你是玩不过我的！";
			return 0; 
		}
	} 
	return 0;
}
